def diss(x,y):
  a = 8
  b = 0
  for col in ["bap_tilang","stop_operasi","bap_polisi","stop_operasi_polisi","penderekan","ocp_roda_dua","ocp_roda_empat","angkut_motor"]:
    if p[col][x] == p[col][y]:
      b+=1
    return (a-b)/a

    for col in [hasil1, hasil2]:
        result = math.pow(abs(col[a] - col[b]), c)
    return result**(1/c)
